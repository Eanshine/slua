
function ByteArrayToLString(buffer)
	return Slua.ToString(buffer:GetData())
end

function LStringToByteArray(buffer)
	buffer = Slua.ToBytes(buffer)
	buffer = Slua.ByteArray(buffer)
	return buffer
end
